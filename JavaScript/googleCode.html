Select logo

var logo = document.querySelector("#hplogo");

Update the source of the logo

logo.setAttribute("srcset", "https://i.pinimg.com/originals/b0/de/52/b0de529ce2af2fc3bc179901451934ae.jpg");

Select all links

var links = document.getElementsByTagName("a");

Change style for all the links

for (var i = 0; i < links.length; i++) {
links[i].style.border = "1px dashed red";
links[i].style.color = "orange";
}

Change all the links

for (var i = 0; i < links.length; i++) {
links[i].setAttribute("href", "https://www.facebook.com");
}
